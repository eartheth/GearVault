Required Database Schema DDL

If you haven't created the tables in your Supabase project, execute this SQL code in the SQL Editor tab of your Supabase dashboard: 

-- Create Cases Table
CREATE TABLE public.cases (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    description TEXT,
    created_at TIMESTAMPTZ DEFAULT now()
);

-- Create Items Table
CREATE TABLE public.items (
    id TEXT PRIMARY KEY, -- Barcode or Serial number
    name TEXT NOT NULL,
    case_id TEXT REFERENCES public.cases(id) ON DELETE SET NULL,
    layer TEXT NOT NULL, -- e.g., 'Top Layer', 'Bottom Layer', 'Main Compartment'
    status TEXT NOT NULL DEFAULT 'Present', -- 'Present', 'Missing', 'Maintenance'
    serial_number TEXT,
    notes TEXT,
    updated_at TIMESTAMPTZ DEFAULT now()
);

-- Create Maintenance Table
CREATE TABLE public.maintenance (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    item_id TEXT REFERENCES public.items(id) ON DELETE CASCADE,
    item_name TEXT NOT NULL,
    task TEXT NOT NULL,
    status TEXT NOT NULL, -- 'OVERDUE', 'DUE SOON', 'SCHEDULED'
    due_date DATE,
    created_at TIMESTAMPTZ DEFAULT now()
);

-- Create Scan History Table
CREATE TABLE public.scan_history (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    timestamp TIMESTAMPTZ DEFAULT now(),
    case_id TEXT REFERENCES public.cases(id),
    case_name TEXT,
    status TEXT, -- 'ALL PRESENT', '1 MISSING', etc.
    operator TEXT DEFAULT 'SeminarDD'
);
